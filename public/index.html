/** * LLM Chat Application Template with D1 Persistence * * Handles chat requests and stores/retrieves messages from D1. */ import { Env, ChatMessage } from "./types"; // Workers AI model ID const MODEL_ID = "@cf/meta/llama-3.3-70b-instruct-fp8-fast"; const SYSTEM_PROMPT = "You are a helpful, friendly assistant. Provide concise and accurate responses."; export default { async fetch(request: Request, env: Env, ctx: ExecutionContext): Promise { const url = new URL(request.url); // Serve frontend static assets if (url.pathname === "/" || !url.pathname.startsWith("/api/")) { return env.ASSETS.fetch(request); } // Chat API routes if (url.pathname === "/api/chat") { if (request.method === "POST") { return handleChatRequest(request, env); } if (request.method === "GET") { const sessionId = url.searchParams.get("sessionId") || "default-session"; const history = await getChatHistoryFromD1(env, sessionId); return new Response(JSON.stringify(history), { headers: { "content-type": "application/json" }, }); } return new Response("Method not allowed", { status: 405 }); } return new Response("Not found", { status: 404 }); }, } satisfies ExportedHandler; /** * Save a message to D1 */ async function saveMessageToD1(env: Env, sessionId: string, role: string, content: string) { await env.D1_DB.prepare( `INSERT INTO chats (session_id, role, content) VALUES (?, ?, ?)` ).bind(sessionId, role, content).run(); } /** * Retrieve chat history from D1 */ async function getChatHistoryFromD1(env: Env, sessionId: string) { const result = await env.D1_DB.prepare( `SELECT role, content, timestamp FROM chats WHERE session_id = ? ORDER BY timestamp ASC` ).bind(sessionId).all(); return result.results; // [{role, content, timestamp}, ...] } /** * Handle chat POST requests */ async function handleChatRequest(request: Request, env: Env): Promise { try { const { sessionId = "default-session", messages: clientMessages = [] } = await request.json(); // Persist user messages for (const msg of clientMessages) { if (msg.role !== "system") { await saveMessageToD1(env, sessionId, msg.role, msg.content); } } // Ensure system prompt if (!clientMessages.some((msg) => msg.role === "system")) { clientMessages.unshift({ role: "system", content: SYSTEM_PROMPT }); } // Call Workers AI model const aiResponse = await env.AI.run(MODEL_ID, { messages: clientMessages, max_tokens: 1024, }, { returnRawResponse: true }); // Stream text const reader = aiResponse.body.getReader(); const decoder = new TextDecoder(); let assistantText = ""; while (true) { const { done, value } = await reader.read(); if (done) break; const chunk = decoder.decode(value, { stream: true }); assistantText += chunk; } // Save AI response to D1 await saveMessageToD1(env, sessionId, "assistant", assistantText); return new Response(assistantText, { status: 200 }); } catch (err) { console.error(err); return new Response(JSON.stringify({ error: "Failed to process request" }), { status: 500, headers: { "content-type": "application/json" }, }); } }
